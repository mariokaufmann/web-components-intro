JSX Expressions

{} erlauben im Markup eingebettete Expressions

            
                function render() {
                    const name = "Peter Parker";
                    return (
                      <p>Hello from {name}!</p>
                    );
                }
            
        
            
                <p>Hello from Peter Parker!</p>

JSX Expressions

            
                  function getName(): string {
                    return "Peter Parker";
                  }

                  function render() {
                    return (
                      <p>Hello from {this.getName()}!</p>
                    );
                  }
            
        
            
                <p>Hello from Peter Parker!</p>

JSX Attribute

            
                  function getImgSrc(): string {
                    return "/img/portrait.jpg";
                  }

                  function render() {
                    return (
                      <img class="portrait" src={this.getImgSrc()}></img>
                    );
                  }
            
    
            
                <img class="portrait" src="/img/portrait.jpg"></img>

JSX conditionals

            
                  function render() {
                    const unreadMessages = ...;
                    return (
                        <h1>Hello User!</h1>
                        {
                            unreadMessages.length > 0 &&
                            <p>
                                You have {unreadMessages.length} messages!
                            </p>
                        }
                    );
                  }
            
    
            
                <h1>Hello User!</h1>
                <p>You have 3 messages!</p>

JSX conditionals

            
                  function render() {
                    const unreadMessages = ...;
                    return (
                        <h1>Hello User!</h1>
                        {
                            unreadMessages.length > 0
                            ?
                            <p>
                                You have {unreadMessages.length} messages!
                            </p>
                            :
                            <p>
                                You have no unread messages!
                            </p>
                        }
                    );
                  }

JSX Listen

            
                  function render() {
                    const numbers = [1, 2, 3];
                    return (
                        <ul>
                            {
                                numbers.map(number => <li>{number}</li>)
                            }
                        </ul>
                    );
                  }
            
    
            
                <ul>
                    <li>1</li>
                    <li>2</li>
                    <li>3</li>
                </ul>
